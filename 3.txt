class Тварина:
    def __init__(self, вид, ім_я):
        self.вид = вид
        self.ім_я = ім_я

class Зоопарк:
    def __init__(self):
        self.тварини = []

    def додати_тварину(self, тварина):
        self.тварини.append(тварина)

    def вилучити_тварину(self, ім_я_тварини):
        for тварина in self.тварини:
            if тварина.ім_я == ім_я_тварини:
                self.тварини.remove(тварина)
                print(f"Тварину {ім_я_тварини} вилучено зі зоопарку.")
                return
        print(f"Тварина з іменем {ім_я_тварини} не знайдена у зоопарку.")

зоопарк = Зоопарк()

тварина1 = Тварина("Лев", "Симба")
тварина2 = Тварина("Зебра", "Марта")

зоопарк.додати_тварину(тварина1)
зоопарк.додати_тварину(тварина2)

зоопарк.вилучити_тварину("Симба")

print("Залишені тварини в зоопарку:")
for тварина in зоопарк.тварини:
    print(тварина.ім_я, "-", тварина.вид)
