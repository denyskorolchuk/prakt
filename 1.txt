class ЗвичайнийДріб:
    def __init__(self, чисельник, знаменник):
        self.чисельник = чисельник
        self.знаменник = знаменник

    def чи_правильний(self):
        return self.чисельник < self.знаменник

# Приклад використання
дріб1 = ЗвичайнийДріб(2, 3)
дріб2 = ЗвичайнийДріб(5, 2)

print("Дріб 1:", "Правильний" if дріб1.чи_правильний() else "Неправильний")
print("Дріб 2:", "Правильний" if дріб2.чи_правильний() else "Неправильний")
