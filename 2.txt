class Вектор:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def додавання(self, інший_вектор):
        новий_x = self.x + інший_вектор.x
        новий_y = self.y + інший_вектор.y
        return Вектор(новий_x, новий_y)

    def віднімання(self, інший_вектор):
        новий_x = self.x - інший_вектор.x
        новий_y = self.y - інший_вектор.y
        return Вектор(новий_x, новий_y)

    def __str__(self):
        return f"({self.x}, {self.y})"

вектор1 = Вектор(3, 5)
вектор2 = Вектор(1, 2)

сума = вектор1.додавання(вектор2)
різниця = вектор1.віднімання(вектор2)

print("Сума:", сума)
print("Різниця:", різниця)
